function drawGauge(id,val,min,max,color,warn,danger){
  const c=document.getElementById(id); if(!c) return;
  const w=130,h=75; c.width=w; c.height=h;
  const ctx=c.getContext('2d');
  ctx.clearRect(0,0,w,h);
  const cx=w/2,cy=h-4,r=56;
  const SA=Math.PI,EA=2*Math.PI;
  const pct=Math.min(Math.max((val-min)/(max-min),0),1);
  const vA=SA+pct*(EA-SA);
  ctx.beginPath();ctx.arc(cx,cy,r,SA,EA);ctx.strokeStyle='#1e2533';ctx.lineWidth=10;ctx.lineCap='round';ctx.stroke();
  function arc(s,e,col){ctx.beginPath();ctx.arc(cx,cy,r,s,e);ctx.strokeStyle=col;ctx.lineWidth=10;ctx.lineCap='butt';ctx.stroke();}
  if(warn!=null&&danger!=null){
    const wA=SA+((warn-min)/(max-min))*(EA-SA);
    const dA=SA+((danger-min)/(max-min))*(EA-SA);
    arc(SA,wA,'#1e3a1e');arc(wA,dA,'#2a2010');arc(dA,EA,'#2a1010');
  }
  const arcCol=danger!=null&&val>=danger?'#ef4444':warn!=null&&val>=warn?'#f97316':color;
  ctx.beginPath();ctx.arc(cx,cy,r,SA,vA);ctx.strokeStyle=arcCol;ctx.lineWidth=10;ctx.lineCap='round';ctx.stroke();
  const na=SA+pct*(EA-SA);
  const nx=cx+Math.cos(na)*(r-14),ny=cy+Math.sin(na)*(r-14);
  ctx.beginPath();ctx.arc(nx,ny,4,0,2*Math.PI);ctx.fillStyle='#fff';ctx.fill();
  ctx.beginPath();ctx.arc(cx,cy,5,0,2*Math.PI);ctx.fillStyle='#2a3548';ctx.fill();
  ctx.beginPath();ctx.arc(cx,cy,3,0,2*Math.PI);ctx.fillStyle='#4a5568';ctx.fill();
}

function createDrops(){
  const wrap=document.getElementById('rainDrops');
  wrap.innerHTML='';
  for(let i=0;i<18;i++){
    const d=document.createElement('div');
    d.className='drop';

    d.style.cssText=`left:${Math.random()*100}%;height:${h}px;animation-duration:${Math.random()*.5+.4}s;animation-delay:-${Math.random()*.8}s;opacity:${Math.random()*.5+.3}`;
    wrap.appendChild(d);
  }
}

function setRainUI(pct, isRaining){
  const card=document.getElementById('rainCard');
  const drops=document.getElementById('rainDrops');
  const badge=document.getElementById('rainBadge');
  const bar=document.getElementById('rainBar');
  const numEl=document.getElementById('valRain');

  numEl.textContent=pct.toFixed(1);
  bar.style.width=pct+'%';

  // bar color by intensity
  const barColor = pct<30?'#0ea5e9': pct<60?'#38bdf8': pct<80?'#7dd3fc':'#bfdbfe';
  bar.style.background=barColor;

  if(isRaining){
    card.className='rain-card card raining';
    drops.className='rain-drops active';
    badge.className='rain-state-badge rain';
    badge.textContent='🌧️ RAINING';
    // rain alert
    const ra=document.getElementById('rainAlert');
    document.getElementById('rainAlertMsg').textContent=`ตรวจพบฝน! ความเข้ม ${pct.toFixed(0)}%`;
    ra.classList.add('show');
    setTimeout(()=>ra.classList.remove('show'),5000);
  } else {
    card.className='rain-card card';
    drops.className='rain-drops';
    badge.className='rain-state-badge dry';
    badge.textContent='☀️ DRY';
  }
}

function addLog(topic,payload,cls){
  const list=document.getElementById('logList');
  if(list.querySelector('[style*="text-align"]')) list.innerHTML='';
  const item=document.createElement('div'); item.className='log-item log-new';

function drawSpark(id,data,color){
  const c=document.getElementById(id);if(!c)return;
  const w=c.offsetWidth||280,h=58;c.width=w;c.height=h;
  const ctx=c.getContext('2d');ctx.clearRect(0,0,w,h);
  if(data.length<2)return;
  const mn=Math.min(...data)-2,mx=Math.max(...data)+2,rng=mx-mn||1;
  ctx.beginPath();ctx.strokeStyle=color;ctx.lineWidth=1.5;
  data.forEach((v,i)=>{const x=(i/(data.length-1))*w,y=h-((v-mn)/rng)*(h-8)-4;i===0?ctx.moveTo(x,y):ctx.lineTo(x,y);});
  ctx.stroke();ctx.lineTo(w,h);ctx.lineTo(0,h);ctx.closePath();ctx.fillStyle=color+'18';ctx.fill();
}

function toggleAuto(){
  autoEnabled=!autoEnabled;
  const btn=document.getElementById('autoToggle'),lbl=document.getElementById('autoLbl'),stat=document.getElementById('autoStat');
  if(autoEnabled){btn.classList.add('active');lbl.textContent='Auto Control: ON';stat.style.color='#60a5fa';runAuto();}
  else{btn.classList.remove('active');lbl.textContent='Auto Control: OFF';stat.textContent='Auto OFF';stat.style.color='#4a5568';}
}
function runAuto(){
  if(!autoEnabled) return;
  const tf=parseFloat(document.getElementById('threshFan').value)||30;
  const tl=parseFloat(document.getElementById('threshLight').value)||500;
  const stat=document.getElementById('autoStat');let actions=[];
  if(lastTemp!==null){
    if(lastTemp>=tf&&!fanOn){setFan(true,true);actions.push(`พัดลม ON (${lastTemp.toFixed(1)}°C ≥ ${tf}°C)`);}
    else if(lastTemp<tf-2&&fanOn){setFan(false,true);actions.push(`พัดลม OFF`);}
  }
  if(lastLight!==null){
    if(lastLight>=tl&&lightOn){setLight(false,true);actions.push(`ไฟ OFF (${Math.round(lastLight)} lux)`);}
    else if(lastLight<tl*.7&&!lightOn){setLight(true,true);actions.push(`ไฟ ON`);}
  }
  if(actions.length){stat.textContent='✓ '+actions[0];stat.style.color='#00d68f';actions.forEach(a=>addLog('🤖 auto',a,''));}
  else{stat.textContent=lastTemp!==null?`${lastTemp.toFixed(1)}°C — ปกติดี`:'รอข้อมูล...';stat.style.color='#00d68f';}
}

const topicMap={
  'sensors/dht/temperature': v=>{
    lastTemp=parseFloat(v);const t=lastTemp;
    updateGauge('numTemp','subTemp','gTemp',t,0,50,'#f97316',35,40,t>40?'⚠️ ร้อนมาก':t>35?'🌡️ อุ่น':'✓ ปกติ',t>40);
    addHist(histTemp,t);drawSpark('cTemp',histTemp,'#f97316');
    if(autoEnabled) runAuto();
  },
  'sensors/dht/humidity': v=>{
    const h=parseFloat(v);
    updateGauge('numHumid','subHumid','gHumid',h,0,100,'#3b82f6',70,90,h>90?'💧 ชื้นมาก':h>70?'💧 ค่อนข้างชื้น':'✓ ปกติ',h>90);
    addHist(histHumid,h);drawSpark('cHumid',histHumid,'#3b82f6');
  },
  'sensors/light/lux': v=>{
    lastLight=parseFloat(v);const l=lastLight;
    updateGauge('numLight','subLight','gLight',l,0,1000,'#eab308',700,900,l>900?'☀️ สว่างมาก':l<100?'🌙 มืด':'✓ พอดี',false);
    if(autoEnabled) runAuto();
  },
  'sensors/bmp/pressure': v=>updateSmall('valPres','subPres',parseFloat(v),'✓ ปกติ',false),
  'sensors/air/co2': v=>{const c=parseFloat(v);updateSmall('valCO2','subCO2',c,c>1000?'⚠️ CO₂ สูง':'✓ ปกติ',c>1000);},
  'sensors/soil/moisture': v=>{const s=parseFloat(v);updateSmall('valSoil','subSoil',s,s<20?'🏜️ แห้งมาก':s>80?'💦 ชุ่มมาก':'✓ ปกติ',false);},
  'sensors/rain/intensity': v=>{
    const pct=parseFloat(v);
    // จะ raining ถ้า intensity > 20% (ปรับได้)
    setRainUI(pct, pct>20);
    addLog('sensors/rain/intensity',`${pct.toFixed(1)}%`,'rain');
  },
  'sensors/rain/detected': v=>{
    const isR=(v==='RAIN');
    // ถ้ายังไม่มี intensity ค่า แต่ได้รับ digital signal
    const curPct=parseFloat(document.getElementById('valRain').textContent)||0;
    if(isR&&curPct<20) setRainUI(50,true);
    addLog('sensors/rain/detected',v,'rain');
  },
  'sensors/alert/flood_risk': v=>{
    showAlert('⚠️ เสี่ยงน้ำท่วม! ฝนตก + ดินชื้นสูง');
  },
  'control/light': v=>{const on=v==='ON'||v==='1';if(on!==lightOn)setLight(on,true);},
  'control/fan':   v=>{const on=v==='ON'||v==='1';if(on!==fanOn)setFan(on,true);}
};

function handleMsg(topic,raw){
  // Rain topics already logged in topicMap, others log here
  if(!topic.startsWith('sensors/rain')) addLog(topic,raw,'');
  let val;
  try{const p=JSON.parse(raw);val=p.value??p.val??p.data??Object.values(p)[0];}catch{val=raw.trim();}
  if(topicMap[topic]) topicMap[topic](isNaN(parseFloat(val))?val:parseFloat(val));
}

document.getElementById('btnConnect').addEventListener('click',()=>{
  if(client){client.end();client=null;setConn('disconnected');document.getElementById('btnConnect').textContent='Connect';return;}
  if(demoInterval){clearInterval(demoInterval);demoInterval=null;document.getElementById('btnDemo').textContent='▶ Demo';}
  const broker=document.getElementById('cfgBroker').value;
  const cid=document.getElementById('cfgClientId').value||'iotdash_'+Date.now();
  const prefix=document.getElementById('cfgPrefix').value;
  document.getElementById('brokerInfo').textContent=broker.replace('wss://','').replace('/mqtt','');
  setConn('connecting');document.getElementById('btnConnect').textContent='Disconnect';
  try{
    client=mqtt.connect(broker,{clientId:cid,clean:true,reconnectPeriod:3000});
    client.on('connect',()=>{setConn('connected');client.subscribe(prefix,{qos:1});client.subscribe('control/#',{qos:1});});
    client.on('message',(t,m)=>handleMsg(t,m.toString()));
    client.on('error',()=>setConn('disconnected'));
    client.on('offline',()=>setConn('disconnected'));
  }catch(e){setConn('disconnected');document.getElementById('btnConnect').textContent='Connect';}
});

  const SA=Math.PI,EA=2*Math.PI;
  const pct=Math.min(Math.max((val-min)/(max-min),0),1);
  const vA=SA+pct*(EA-SA);
  ctx.beginPath();ctx.arc(cx,cy,r,SA,EA);ctx.strokeStyle='#1e2533';ctx.lineWidth=10;ctx.lineCap='round';ctx.stroke();
  function arc(s,e,col){ctx.beginPath();ctx.arc(cx,cy,r,s,e);ctx.strokeStyle=col;ctx.lineWidth=10;ctx.lineCap='butt';ctx.stroke();}
  if(warn!=null&&danger!=null){
    const wA=SA+((warn-min)/(max-min))*(EA-SA);
    const dA=SA+((danger-min)/(max-min))*(EA-SA);
    arc(SA,wA,'#1e3a1e');arc(wA,dA,'#2a2010');arc(dA,EA,'#2a1010');
  }
  const arcCol=danger!=null&&val>=danger?'#ef4444':warn!=null&&val>=warn?'#f97316':color;
  ctx.beginPath();ctx.arc(cx,cy,r,SA,vA);ctx.strokeStyle=arcCol;ctx.lineWidth=10;ctx.lineCap='round';ctx.stroke();
  const na=SA+pct*(EA-SA);
  const nx=cx+Math.cos(na)*(r-14),ny=cy+Math.sin(na)*(r-14);
  ctx.beginPath();ctx.arc(nx,ny,4,0,2*Math.PI);ctx.fillStyle='#fff';ctx.fill();
  ctx.beginPath();ctx.arc(cx,cy,5,0,2*Math.PI);ctx.fillStyle='#2a3548';ctx.fill();
  ctx.beginPath();ctx.arc(cx,cy,3,0,2*Math.PI);ctx.fillStyle='#4a5568';ctx.fill();
}
